Get ID Value
    [Arguments]    ${URL}
    @{URL_PARAMS}=    Split String    ${URL}    ?
    ${PARAM_LINE}=    Get From List    ${URL_PARAMS}    1
    @{PARAMS}=    Split String    ${PARAM_LINE}    &
    ${ID_PARAM}=    Get From List    ${PARAMS}    0
    @{NAME_VALUE}    Split String    ${ID_PARAM}    =
    ${VALUE}=    Get From List    ${NAME_VALUE}    1
    [Return]    ${VALUE}
